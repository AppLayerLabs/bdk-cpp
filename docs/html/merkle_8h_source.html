43 template <typename TxType> Merkle(const std::vector<TxType>& txs) {
44 // Mount the base leaves
45 std::vector<Hash> tmp;
46 for (auto tx : txs) {
47 tmp.emplace_back(std::move(Utils::sha3(tx.hash().get())));
48 }
49 this->tree.emplace_back(tmp);
50 // Make the layers up to root

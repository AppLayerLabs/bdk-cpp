30 inline void checkKeyAndCopy(const Key& key) const {
31 check();
32 auto itP = mapPtr->find(key);
33 if (itP == mapPtr->end()) {
34 auto itM = map.find(key);

64 inline void checkKeyAndThrow(const Key& key) const {
65 check();
66 auto itP = mapPtr->find(key);
67 if (itP == mapPtr->end()) {
68 auto itM = map.find(key);
69 if (itM == map.end()) {
70 throw std::runtime_error("Key not found");
71 } else {
72 auto itD = erasedKeys->find(key);
73 if (itD == erasedKeys->end()) {

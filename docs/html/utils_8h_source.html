302 template <class T> inline unsigned bytesRequired(T i) {
303 // bigint does not carry sign bit on shift
304 static_assert(std::is_same<bigint, T>::value || !std::numeric_limits<T>::is_signed, "only unsigned types or bigint supported");
305 unsigned ic = 0;
306 for (; i != 0; ++ic, i >>= 8) {}

316 template <class T> std::string uintToBytes(T i) {
317 std::string ret(bytesRequired(i), 0x00);
318 uint8_t* b = reinterpret_cast<uint8_t*>(&ret.back());
319 for (; i; i >>= 8) *(b--) = (uint8_t)(i & 0xff);
